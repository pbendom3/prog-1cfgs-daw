Ejercicio 2

public static void imprimir(int cont, int fin){

        if (cont<=fin) {
            System.out.println(cont);
            cont++;
            imprimir(cont,fin);
        }

    }

    public static void main(String[] args){
        Scanner teclado = new Scanner(System.in);
        System.out.println("Introduce el número N hasta el que quieres contar: ");
        int n = teclado.nextInt();
        imprimir(1,n);
    }

Ejercicio 3

public static void main(String[] args){
        int n = 10;
        System.out.println("Suma = " + contar(1,n));
    }

    public static int contar(int cont, int fin){

        if (cont<fin){
            return cont + contar(cont+1,fin);
        }

        return cont;
    }
