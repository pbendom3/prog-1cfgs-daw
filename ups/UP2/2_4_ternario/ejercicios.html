Solución

//caso 1
int a = 1, b = 2, c = 3;

int resultado = a++ == 2 && c > b++ ? a += b : //b no se incrementa
                    a > 0 && a == b ? a += c :
                                a == c ? a++ : a--; //a=5, b=2, c=3,resultado=5

//caso 2
a = 1; b = 2; c = 3;

resultado = a++ == 1 && a > b++ ? a += b :
                a > 0 && a == b ? a += c :
                            a == c ? a++ : a--; //a=1, b=3, c=3, resultado=2

//caso 3
a = 1; b = 2; c = 3;

resultado = a++ == 1 && a > b++ ? a += b :
                a > 0 && a == b ? a += c :
                            b == c ? a++ : a--; //a=3, b=3, c=3, resultado=2
