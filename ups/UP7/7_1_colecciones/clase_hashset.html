Clase HashSet

HashSet es muy rápida, pero no garantiza el orden de los elementos. Es decir, si en una lista tenemos guardados A, B y C, cada vez que la imprimamos podría mostrarse en cualquier orden.

import java.util.HashSet;
import java.util.Set;

public class EjemploHashSet {

    public static void main(String[] args) {

        Set<String> conjunto = new HashSet<>();

        conjunto.add("B");
        conjunto.add("A");
        conjunto.add("C");
        conjunto.add("A"); // no se añadirá, porque ya existe

        System.out.println("Conjunto: " + conjunto); // [A, B, C] (puede salir en otro orden)

        System.out.println("¿Contiene 'B'?: " + conjunto.contains("B")); // true

        conjunto.remove("A");

        System.out.println("Tamaño del conjunto: " + conjunto.size()); // 2

    }

}

Si necesitamos que se mantenga el orden de inserción, en lugar de HashSet deberemos implementar una lista de tipo LinkedHashSet:

import java.util.LinkedHashSet;
import java.util.Set;

public class EjemploLinkedHashSet {

    public static void main(String[] args) {

        Set<Integer> numeros = new LinkedHashSet<>();

        numeros.add(3);
        numeros.add(1);
        numeros.add(2);

        System.out.println("Conjunto ordenado: " + numeros); // [3, 1, 2]

    }

}

Y si queremos que además los elementos se ordenen automáticamente (ascendentemente) conforme se van añadiendo, entonces necesitaremos una estructura de tipo árbol: TreeSet.

import java.util.TreeSet;
import java.util.Set;

public class EjemploTreeSet {

    public static void main(String[] args) {

        Set<Integer> treeSet = new TreeSet<>();

        treeSet.add(50);
        treeSet.add(10);
        treeSet.add(30);

        System.out.println("Conjunto ordenado: " + treeSet); // [10, 30, 50]

    }

}
Comparación de HashSet, TreeSet y LinkedHashSet

Vamos a añadir los siguientes valores en tres estructuras distintas: HashSet, TreeSet y LinkedHashSet.

String[] palabras = {"banana", "manzana", "pera", "uva", "manzana", "cereza", "banana"};
import java.util.HashSet;
import java.util.LinkedHashSet;
import java.util.Set;
import java.util.TreeSet;

public class ComparacionSets {
    public static void main(String[] args) {

        String[] palabras = {"banana", "manzana", "pera", "uva", "manzana", "cereza", "banana"};

        Set<String> hashSet = new HashSet<>();
        Set<String> treeSet = new TreeSet<>();
        Set<String> linkedHashSet = new LinkedHashSet<>();

        for (String palabra : palabras) {
            hashSet.add(palabra);
            treeSet.add(palabra);
            linkedHashSet.add(palabra);
        }

        System.out.println("HashSet (orden impredecible): " + hashSet);
        System.out.println("TreeSet (ordenado alfabéticamente): " + treeSet);
        System.out.println("LinkedHashSet (orden de inserción): " + linkedHashSet);

    }
}
